pse_model <- readLines("./models/pse-unconstrained.stan") |>
  paste(collapse = "\n")

fle_stan_data <- list(
  N = nrow(fle_data) |> as.integer()
  , n_subjects = length(unique(fle_data$ID)) |> as.integer()
  , id = as.numeric(as.factor(fle_data$ID))
  , y = fle_data$dva

  # Prior settings
  , mu_theta_mean = 1
  , mu_theta_sd = 1
  , pi_mean = -0.5
  , pi_sd = 0.5
)

fle_bayes <- rstan::stan(
  model_code = pse_model
  , data = fle_stan_data
  , iter = 1e5/4 + 1000
  , warmup = 1000
  , chains = 4
  , cores = 4
  , seed = 123
)
